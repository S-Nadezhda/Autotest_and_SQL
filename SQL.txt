Задание 1
SELECT c.login, COUNT(*) FROM "Couriers" AS c INNER JOIN "Orders" AS o ON o."courierId"=c.id WHERE o."inDelivery" = true GROUP BY c.login;

Задание 2
SELECT cancelled, finished, "inDelivery", CASE WHEN cancelled = true THEN -1 WHEN finished = true THEN 2 WHEN "inDelivery" = true THEN 1 ELSE 0 END status FROM "Orders";
